ORG 1000H 	; Datos
;V1 DB 1
;V2 DB 2
CADENA DB "LeandroMatias"
FINC DB ?
POS DB 12
	
ORG 3000H  ; Subrutina
SWAP: PUSH AX; guardo ax para que la subrutina sea reusable
PUSH CX
PUSH BX
MOV CL, [BX]
MOV BX, AX
MOV CH, [BX]
MOV [BX], CL
POP BX
MOV [BX], CH
POP CX
POP AX
RET



ORG 2000H   ; Programa Ppal.
MOV DH, POS
BUCLE1:CMP DH, 0
JZ FIN
MOV BX, OFFSET CADENA
MOV AX, OFFSET CADENA + 1
DEC DH
MOV DL, OFFSET FINC - OFFSET CADENA
DEC DL
BUCLE2: CMP DL, 0
JZ BUCLE1
CALL SWAP
INC BX
INC AX
DEC DL
JMP BUCLE2
FIN:HLT
END
