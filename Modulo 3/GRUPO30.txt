ORG 3000H  ; Subrutina
SWAP: PUSH AX; guardo ax para que la subrutina sea reusable
PUSH CX
PUSH BX
MOV CL, [BX]
MOV BX, AX
MOV CH, [BX]
MOV [BX], CL
POP BX
MOV [BX], CH
POP CX
POP AX
RET

ES_VOCAL: PUSH AX
PUSH BX
MOV BX, SP
ADD BX, 6
MOV DL, [BX]
MOV BX, OFFSET VOCALES
MOV CL, OFFSET FINV - OFFSET VOCALES
REPETIR: CMP CL, 0
JZ FINR
CMP DL, [BX]
JZ VOCAL
MOV CH, 0
INC BX
DEC CL
JMP REPETIR
VOCAL:MOV CH,  0FFH
JMP REPETIR
POP BX
POP BX
POP AX
FINR:RET

BUCLE: CMP AX, BX

CALL SWAP
MOV DH, [BX]

POP DX
